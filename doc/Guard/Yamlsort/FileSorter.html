def self.sort(obj)
  if obj.is_a? Hash
    sorted = {}
    obj.keys.sort.each { |k| sorted[k] = sort(obj[k]) }
    sorted
  else 
    return obj
  end
end

def load_yaml
  self.yaml= File.open(@path, "r") do |f| 
    contents = f.read
    contents.split("\n").each do |line|
      @comments.push(line) if line[0] == '#'
    end
    YAML.load(contents) 
  end
end

def write_yaml
  File.open(@path, "w") do |f| 
    if !@comments.empty?
      f.write @comments.join("\n")
      f.write("\n")
    end
    f.write YAML.dump(yaml)
  end
end
